package osplus.pkptuna.hazelcast.pubsub; import com.hazelcast.core.HazelcastInstance; import com.hazelcast.topic.ITopic; import com.hazelcast.topic.Message; import com.hazelcast.topic.MessageListener; import org.slf4j.Logger; import org.slf4j.LoggerFactory; import osplus.pkptuna.hazelcast.flows.FlowHandler; import osplus.pkptuna.hazelcast.flows.FlowHandlerRegistry; import java.util.List; import java.util.Map; import java.util.UUID; import java.util.concurrent.ConcurrentHashMap; import java.util.concurrent.atomic.AtomicInteger; public class GenericHazelcastRouter { private static final Logger LOGGER = LoggerFactory.getLogger(GenericHazelcastRouter.class); private final HazelcastInstance hazelcastInstance; private final List flowConfigs; private final FlowHandlerRegistry registry; private final Map roundRobinCounters = new ConcurrentHashMap<>(); public GenericHazelcastRouter(HazelcastInstance hazelcastInstance, List flowConfigs, FlowHandlerRegistry registry) { this.hazelcastInstance = hazelcastInstance; this.flowConfigs = flowConfigs; this.registry = registry; } public void initRouters() { if (flowConfigs == null || flowConfigs.isEmpty()) { LOGGER.warn("No flow configurations provided."); return; } for (FlowConfig config : flowConfigs) { String flowName = config.name(); roundRobinCounters.putIfAbsent(flowName, new AtomicInteger(0)); ITopic topic = hazelcastInstance.getReliableTopic(config.topicName()); topic.addMessageListener(new MessageListener<>() { @Override public void onMessage(Message message) { try { FlowHandler handler = registry.getHandler(flowName); if (handler == null) { LOGGER.warn("No handler found for flow: {}", flowName); return; } handler.handleIncomingMessage(message.getMessageObject(), flowName, hazelcastInstance); } catch (Exception ex) { LOGGER.error("Error handling message for flow {}: {}", flowName, ex.getMessage(), ex); } } }); } } public record FlowConfig(String name, String topicName) {} }
