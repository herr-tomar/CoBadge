@Configuration public class FlowHandlerRegistration { @Bean public DisputeLookupFlowHandler disputeLookupFlowHandler(DisputeService service, FlowHandlerRegistry registry) { DisputeLookupFlowHandler handler = new DisputeLookupFlowHandler(service); registry.register("dispute-lookup", handler); return handler; } @Bean public DisputeDetailFlowHandler disputeDetailFlowHandler(DisputeDetailService service, FlowHandlerRegistry registry) { DisputeDetailFlowHandler handler = new DisputeDetailFlowHandler(service); registry.register("dispute-detail", handler); return handler; } }
