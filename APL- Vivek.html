package osplus.pkptuna.mapperutil; import java.time.Instant; import com.google.protobuf.Timestamp; public class TimestampMapper { public Instant map(Timestamp timestamp) { return timestamp == null ? null : Instant.ofEpochSecond(timestamp.getSeconds(), timestamp.getNanos()); } public Timestamp map(Instant instant) { if (instant == null) return null; return Timestamp.newBuilder() .setSeconds(instant.getEpochSecond()) .setNanos(instant.getNano()) .build(); } }
