package osplus.pkptuna.hazelcast.router; import com.hazelcast.core.HazelcastInstance; import com.hazelcast.map.IMap; import com.hazelcast.topic.ITopic; import com.hazelcast.topic.Message; import com.hazelcast.topic.MessageListener; import jakarta.annotation.PostConstruct; import org.slf4j.Logger; import org.slf4j.LoggerFactory; import osplus.apl.core.api.AplComponent; import osplus.pkptuna.hazelcast.configprops.HazelcastFlowConfigs; import osplus.pkptuna.hazelcast.pubsub.flow.FlowHandler; import osplus.pkptuna.hazelcast.pubsub.flow.FlowHandlerRegistry; import java.util.Map; import java.util.UUID; import java.util.concurrent.ConcurrentHashMap; import java.util.concurrent.atomic.AtomicInteger; @AplComponent public class GenericHazelcastRouter { private static final Logger LOGGER = LoggerFactory.getLogger(GenericHazelcastRouter.class); private final HazelcastInstance hazelcastInstance; private final HazelcastFlowConfigs flowConfigs; private final FlowHandlerRegistry handlerRegistry; // Maintain round-robin counters per flow private final Map roundRobinCounters = new ConcurrentHashMap<>(); public GenericHazelcastRouter(HazelcastInstance hazelcastInstance, HazelcastFlowConfigs flowConfigs, FlowHandlerRegistry handlerRegistry) { this.hazelcastInstance = hazelcastInstance; this.flowConfigs = flowConfigs; this.handlerRegistry = handlerRegistry; } @PostConstruct public void initRouters() { for (Map.Entry entry : flowConfigs.getFlowMap().entrySet()) { String flowName = entry.getKey(); HazelcastFlowConfigs.FlowConfig config = entry.getValue(); roundRobinCounters.put(flowName, new AtomicInteger(0)); ITopic topic = hazelcastInstance.getReliableTopic(config.requestTopic()); topic.addMessageListener(new RoutingListener<>(flowName, config)); LOGGER.info("‚úÖ Listening to requestTopic='{}' for flow='{}'.", config.requestTopic(), flowName); } } private class RoutingListener implements MessageListener { private final String flowName; private final HazelcastFlowConfigs.FlowConfig config; RoutingListener(String flowName, HazelcastFlowConfigs.FlowConfig config) { this.flowName = flowName; this.config = config; } @Override public void onMessage(Message message) { try { FlowHandler handler = (FlowHandler) handlerRegistry.getHandler(flowName); T request = handler.deserializeRequest(message.getMessageObject()); String correlationId = handler.extractCorrelationId(request); String issuerId = handler.extractIssuerId(request); // Optional based on logic String lockKey = "route-lock-" + correlationId; IMap routerMap = hazelcastInstance.getMap("router-claim-map"); String uniqueId = UUID.randomUUID().toString(); String existing = routerMap.putIfAbsent(lockKey, uniqueId); if (existing != null) { LOGGER.info("‚õî Skipped duplicate routing for correlationId={}", correlationId); return; } int partition = roundRobinCounters.get(flowName).getAndIncrement() % config.partitions(); String mapKey = config.mapBase() + partition; hazelcastInstance.getMap(mapKey).put(correlationId, request); LOGGER.info("üì® Routed correlationId='{}' to map '{}'", correlationId, mapKey); } catch (Exception e) { LOGGER.error("‚ùå Routing failure for flow '{}': {}", flowName, e.getMessage(), e); } } } }
