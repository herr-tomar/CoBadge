public class ProtoParserUtil { public static List parseAll(File protoDir) { List messages = new ArrayList<>(); File[] protoFiles = protoDir.listFiles((dir, name) -> name.endsWith(".proto")); if (protoFiles == null) return messages; for (File protoFile : protoFiles) { messages.addAll(parse(protoFile)); } return messages; } public static List parse(File protoFile) { List messageModels = new ArrayList<>(); try (BufferedReader reader = new BufferedReader(new FileReader(protoFile))) { String line; ProtoMessageModel currentMessage = null; while ((line = reader.readLine()) != null) { line = line.trim(); // Start of message if (line.startsWith("message ")) { String messageName = line.split("\\s+")[1].replace("{", "").trim(); currentMessage = new ProtoMessageModel(messageName); continue; } // End of message if (line.startsWith("}")) { if (currentMessage != null) { messageModels.add(currentMessage); currentMessage = null; } continue; } // Field line if (currentMessage != null && !line.isEmpty() && line.contains(";")) { String[] parts = line.split("\\s+"); if (parts.length >= 3) { String type = parts[0]; String name = parts[1]; if (name.endsWith(";")) name = name.substring(0, name.length() - 1); currentMessage.addField(new ProtoFieldModel(type, name)); } } } } catch (Exception e) { System.err.println("Failed to parse proto file: " + protoFile.getName()); e.printStackTrace(); } return messageModels;
