package osplus.pkptuna.protobuf.plugin; import java.io.File; import java.io.IOException; import java.util.List; import freemarker.template.Configuration; import freemarker.template.Template; import freemarker.template.TemplateExceptionHandler; import freemarker.template.Version; public class RestDtoGeneratorPlugin { public static void main(String[] args) { if (args.length < 2) { System.err.println("Usage: RestDtoGeneratorPlugin "); System.exit(1); } File protoDir = new File(args[0]); File outputDir = new File(args[1]); if (!protoDir.exists() || !protoDir.isDirectory()) { System.err.println("Invalid proto input directory: " + protoDir.getAbsolutePath()); System.exit(2); } outputDir.mkdirs(); try { // 1. Set up Freemarker Configuration cfg = new Configuration(new Version("2.3.32")); cfg.setClassForTemplateLoading(RestDtoGeneratorPlugin.class, "/templates"); cfg.setDefaultEncoding("UTF-8"); cfg.setTemplateExceptionHandler(TemplateExceptionHandler.RETHROW_HANDLER); // 2. For each .proto file, parse and generate DTOs (pseudo code) List messageModels = ProtoParserUtil.parseAll(protoDir); for (ProtoMessageModel model : messageModels) { Template template = cfg.getTemplate("rest-dto.ftl"); File outFile = new File(outputDir, model.getClassName() + ".java"); TemplateWriterUtil.writeTemplate(template, model, outFile); } System.out.println("REST DTO generation completed successfully."); } catch (IOException e) { e.printStackTrace(); System.exit(3); } } }
