@Bean public List hazelcastFlowConfigs(HazelcastFlowConfigs external) { List flowConfigList = external.getFlows().stream() .map(c -> new FlowConfig( c.getName(), c.getRequestTopic(), c.getResponseTopic(), c.getMapBase(), c.getPartitions() )) .collect(Collectors.toList()); LOGGER.info("Loaded hazelcast flow configs: {}", flowConfigList); return flowConfigList; }
