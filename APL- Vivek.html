package osplus.pkptuna.hazelcast.pubsub; import com.hazelcast.core.HazelcastInstance; import org.slf4j.Logger; import org.slf4j.LoggerFactory; import org.springframework.scheduling.TaskScheduler; import java.time.Duration; import java.util.List; import java.util.Map; import java.util.concurrent.ConcurrentHashMap; /** * Generic consumer that polls Hazelcast maps based on flow configuration. * Flow-specific logic is injected via FlowConfigProvider and FlowHandlerResolver interfaces. */ public class GenericHazelcastConsumer { private static final Logger LOGGER = LoggerFactory.getLogger(GenericHazelcastConsumer.class); private final HazelcastInstance hazelcastInstance; private final TaskScheduler taskScheduler; private final FlowConfigProvider configProvider; private final FlowHandlerResolver handlerResolver; private final Map partitionTracker = new ConcurrentHashMap<>(); public GenericHazelcastConsumer(HazelcastInstance hazelcastInstance, TaskScheduler taskScheduler, FlowConfigProvider configProvider, FlowHandlerResolver handlerResolver) { this.hazelcastInstance = hazelcastInstance; this.taskScheduler = taskScheduler; this.configProvider = configProvider; this.handlerResolver = handlerResolver; } public void startPolling() { taskScheduler.scheduleAtFixedRate(this::pollRequestMaps, Duration.ofMillis(1000)); } public void pollRequestMaps() { List configs = configProvider.getFlowConfigs(); if (configs == null || configs.isEmpty()) { LOGGER.warn("No flows configured. Skipping poll."); return; } for (FlowConfig config : configs) { FlowHandler handler = handlerResolver.resolve(config.getName()); if (handler == null) { LOGGER.warn("No handler found for flow: {}", config.getName()); continue; } handler.handle(hazelcastInstance, config); } } // --- Supporting Interfaces --- public interface FlowConfigProvider { List getFlowConfigs(); } public interface FlowHandlerResolver { FlowHandler resolve(String flowName); } public interface FlowHandler { void handle(HazelcastInstance instance, FlowConfig config); } public static class FlowConfig { private final String name; private final String mapName; public FlowConfig(String name, String mapName) { this.name = name; this.mapName = mapName; } public String getName() { return name; } public String getMapName() { return mapName; } } }
