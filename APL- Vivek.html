package osplus.pkptuna.hazelcast.pubsub; import com.hazelcast.core.HazelcastInstance; import com.hazelcast.map.IMap; import com.hazelcast.topic.ITopic; import com.hazelcast.topic.Message; import com.hazelcast.topic.MessageListener; import org.slf4j.Logger; import org.slf4j.LoggerFactory; import osplus.pkptuna.hazelcast.configprops.HazelcastFlowConfigs; import osplus.pkptuna.hazelcast.flows.FlowHandler; import osplus.pkptuna.hazelcast.flows.FlowHandlerRegistry; import java.util.List; import java.util.Map; import java.util.UUID; import java.util.concurrent.ConcurrentHashMap; import java.util.concurrent.atomic.AtomicInteger; public class GenericHazelcastRouter { private static final Logger LOGGER = LoggerFactory.getLogger(GenericHazelcastRouter.class); private final HazelcastInstance hazelcastInstance; private final HazelcastFlowConfigs flowConfigs; private final FlowHandlerRegistry registry; private final Map roundRobinCounters = new ConcurrentHashMap<>(); public GenericHazelcastRouter(HazelcastInstance hazelcastInstance, HazelcastFlowConfigs flowConfigs, FlowHandlerRegistry registry) { this.hazelcastInstance = hazelcastInstance; this.flowConfigs = flowConfigs; this.registry = registry; } public void initRouters() { List flowList = flowConfigs.flows(); if (flowList == null || flowList.isEmpty()) { LOGGER.warn("No Hazelcast flow configurations found."); return; } for (HazelcastFlowConfigs.FlowConfig config : flowList) { String flowName = config.name(); roundRobinCounters.putIfAbsent(flowName, new AtomicInteger(0)); ITopic topic = hazelcastInstance.getReliableTopic(config.requestTopic()); topic.addMessageListener(new MessageListener<>() { @Override public void onMessage(Message message) { try { FlowHandler handler = (FlowHandler) registry.getHandler(flowName); if (handler == null) { LOGGER.warn("No handler found for flow: {}", flowName); return; } Object request = handler.deserializeRequest(message.getMessageObject()); String correlationId = handler.extractCorrelationId(request); Object response = handler.handleRequest(request); Object wrapped = handler.wrapResponse(correlationId, response); int partition = getNextPartition(flowName, config.partitions()); String mapName = config.mapBase() + "-" + partition; IMap map = hazelcastInstance.getMap(mapName); map.set(UUID.randomUUID(), wrapped); LOGGER.debug("Published response to map: {}", mapName); } catch (Exception ex) { LOGGER.error("Error handling message for flow {}: {}", flowName, ex.getMessage(), ex); } } }); } } private int getNextPartition(String flowName, int maxPartitions) { AtomicInteger counter = roundRobinCounters.get(flowName); int next = counter.getAndUpdate(prev -> (prev + 1) % maxPartitions); return next; } }
